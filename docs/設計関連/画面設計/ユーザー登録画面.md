### SignupView.vue
```vue
<template>
  <div class="signup-view">
    <AppHeader title="家計簿アプリ" />
    
    <v-container class="fill-height">
      <v-row justify="center" align="center">
        <v-col cols="12" sm="8" md="6" lg="5">
          <v-card class="pa-6">
            <v-card-title class="text-center mb-4">
              <h2>ユーザー登録</h2>
            </v-card-title>
            
            <v-card-text>
              <SignupForm @signup="handleSignup" :loading="isLoading" />
            </v-card-text>
            
            <v-card-actions class="justify-center">
              <LoginLink />
            </v-card-actions>
          </v-card>
        </v-col>
      </v-row>
    </v-container>
  </div>
</template>

<script setup lang="ts">
import { ref } from 'vue'
import { useRouter } from 'vue-router'
import { useAuthStore } from '@/stores/auth'
import { SignupRequest } from '@/types/auth'

const router = useRouter()
const authStore = useAuthStore()
const isLoading = ref(false)

const handleSignup = async (userData: SignupRequest) => {
  try {
    isLoading.value = true
    await authStore.signup(userData)
    router.push('/dashboard')
  } catch (error) {
    console.error('Signup failed:', error)
  } finally {
    isLoading.value = false
  }
}
</script>
```

### SignupForm.vue
```vue
<template>
  <v-form @submit.prevent="onSubmit" ref="formRef">
    <!-- ユーザー名 -->
    <v-text-field
      v-model="form.username"
      label="ユーザー名"
      :rules="usernameRules"
      required
      prepend-inner-icon="mdi-account"
      variant="outlined"
      class="mb-3"
      hint="3文字以上20文字以下で入力してください"
    />
    
    <!-- メールアドレス -->
    <v-text-field
      v-model="form.email"
      label="メールアドレス"
      type="email"
      :rules="emailRules"
      required
      prepend-inner-icon="mdi-email"
      variant="outlined"
      class="mb-3"
    />
    
    <!-- パスワード -->
    <v-text-field
      v-model="form.password"
      label="パスワード"
      :type="showPassword ? 'text' : 'password'"
      :rules="passwordRules"
      required
      prepend-inner-icon="mdi-lock"
      :append-inner-icon="showPassword ? 'mdi-eye' : 'mdi-eye-off'"
      @click:append-inner="showPassword = !showPassword"
      variant="outlined"
      class="mb-3"
      hint="8文字以上、英数字を含む"
    />
    
    <!-- パスワード確認 -->
    <v-text-field
      v-model="form.passwordConfirm"
      label="パスワード確認"
      :type="showPasswordConfirm ? 'text' : 'password'"
      :rules="passwordConfirmRules"
      required
      prepend-inner-icon="mdi-lock-check"
      :append-inner-icon="showPasswordConfirm ? 'mdi-eye' : 'mdi-eye-off'"
      @click:append-inner="showPasswordConfirm = !showPasswordConfirm"
      variant="outlined"
      class="mb-4"
    />
    
    <!-- 利用規約同意 -->
    <v-checkbox
      v-model="form.agreeToTerms"
      :rules="termsRules"
      required
      class="mb-4"
    >
      <template #label>
        <span>
          <a href="/terms" target="_blank" class="text-primary">利用規約</a>
          に同意します
        </span>
      </template>
    </v-checkbox>
    
    <!-- 登録ボタン -->
    <v-btn
      type="submit"
      color="primary"
      size="large"
      block
      :loading="loading"
      :disabled="!isFormValid"
    >
      登録する
    </v-btn>
  </v-form>
</template>

<script setup lang="ts">
import { ref, computed } from 'vue'
import { SignupRequest } from '@/types/auth'

interface Props {
  loading?: boolean
}

interface Emits {
  (e: 'signup', userData: SignupRequest): void
}

const props = withDefaults(defineProps<Props>(), {
  loading: false
})

const emit = defineEmits<Emits>()

const formRef = ref()
const showPassword = ref(false)
const showPasswordConfirm = ref(false)

const form = ref({
  username: '',
  email: '',
  password: '',
  passwordConfirm: '',
  agreeToTerms: false
})

// バリデーションルール
const usernameRules = [
  (v: string) => !!v || 'ユーザー名は必須です',
  (v: string) => (v && v.length >= 3) || 'ユーザー名は3文字以上で入力してください',
  (v: string) => (v && v.length <= 20) || 'ユーザー名は20文字以下で入力してください',
  (v: string) => /^[a-zA-Z0-9_-]+$/.test(v) || '英数字、アンダースコア、ハイフンのみ使用可能です'
]

const emailRules = [
  (v: string) => !!v || 'メールアドレスは必須です',
  (v: string) => /.+@.+\..+/.test(v) || '有効なメールアドレスを入力してください'
]

const passwordRules = [
  (v: string) => !!v || 'パスワードは必須です',
  (v: string) => (v && v.length >= 8) || 'パスワードは8文字以上で入力してください',
  (v: string) => /(?=.*[a-zA-Z])(?=.*[0-9])/.test(v) || '英文字と数字を含める必要があります'
]

const passwordConfirmRules = [
  (v: string) => !!v || 'パスワード確認は必須です',
  (v: string) => v === form.value.password || 'パスワードが一致しません'
]

const termsRules = [
  (v: boolean) => v || '利用規約に同意する必要があります'
]

const isFormValid = computed(() => {
  return form.value.username && 
         form.value.email && 
         form.value.password && 
         form.value.passwordConfirm &&
         form.value.agreeToTerms &&
         usernameRules.every(rule => rule(form.value.username) === true) &&
         emailRules.every(rule => rule(form.value.email) === true) &&
         passwordRules.every(rule => rule(form.value.password) === true) &&
         passwordConfirmRules.every(rule => rule(form.value.passwordConfirm) === true)
})

const onSubmit = async () => {
  const { valid } = await formRef.value.validate()
  if (valid) {
    const { passwordConfirm, agreeToTerms, ...signupData } = form.value
    emit('signup', signupData)
  }
}
</script>
```

## バリデーション仕様

### ユーザー名
- **必須**: ✓
- **文字数**: 3-20文字
- **使用可能文字**: 英数字、アンダースコア(_)、ハイフン(-)
- **重複チェック**: サーバーサイドでリアルタイム確認

### メールアドレス
- **必須**: ✓
- **形式**: RFC準拠
- **重複チェック**: サーバーサイドでリアルタイム確認

### パスワード
- **必須**: ✓
- **文字数**: 8文字以上
- **複雑さ**: 英文字と数字を含む
- **強度表示**: リアルタイムで強度を可視化

## セキュリティ機能

### パスワード強度チェック
```vue
<template>
  <div class="password-strength">
    <v-progress-linear
      :value="passwordStrength.score * 25"
      :color="passwordStrength.color"
      height="4"
      class="mb-2"
    />
    <div class="text-caption" :class="`text-${passwordStrength.color}`">
      {{ passwordStrength.message }}
    </div>
  </div>
</template>

<script setup lang="ts">
import { computed } from 'vue'

const props = defineProps<{
  password: string
}>()

const passwordStrength = computed(() => {
  const password = props.password
  let score = 0
  
  if (password.length >= 8) score++
  if (/[a-z]/.test(password)) score++
  if (/[A-Z]/.test(password)) score++
  if (/[0-9]/.test(password)) score++
  if (/[^a-zA-Z0-9]/.test(password)) score++
  
  const levels = [
    { score: 0, color: 'grey', message: '' },
    { score: 1, color: 'red', message: '弱い' },
    { score: 2, color: 'orange', message: '普通' },
    { score: 3, color: 'yellow', message: '良い' },
    { score: 4, color: 'green', message: '強い' },
    { score: 5, color: 'green', message: '非常に強い' }
  ]
  
  return levels[score]
})
</script>
```

### リアルタイム重複チェック
```typescript
// composables/useValidation.ts
import { ref, watch } from 'vue'
import { debounce } from 'lodash-es'
import { authApi } from '@/api/auth'

export const useEmailValidation = () => {
  const isCheckingEmail = ref(false)
  const emailExists = ref(false)
  
  const checkEmailExists = debounce(async (email: string) => {
    if (!email || !/.+@.+\..+/.test(email)) return
    
    try {
      isCheckingEmail.value = true
      const response = await authApi.checkEmailExists(email)
      emailExists.value = response.exists
    } catch (error) {
      console.error('Email check failed:', error)
    } finally {
      isCheckingEmail.value = false
    }
  }, 500)
  
  return {
    isCheckingEmail,
    emailExists,
    checkEmailExists
  }
}
```

const validationCache = new Map<string, boolean>()

const checkUsernameWithCache = (username: string) => {
  if (validationCache.has(username)) {
    return validationCache.get(username)
  }
