# レポート画面設計

## 画面概要

支出データの分析とレポート機能を提供する画面です。グラフやチャートを用いて視覚的に家計の状況を把握できます。

### ReportView.vue
```vue
<template>
  <div class="report-view">
    <AppHeader title="レポート" />
    
    <v-container class="pa-4">
      <!-- フィルター -->
      <ReportFilters
        v-model:start-date="filters.startDate"
        v-model:end-date="filters.endDate"
        v-model:categories="filters.categories"
        v-model:comparison-period="filters.comparisonPeriod"
        @filter-change="loadReportData"
        @export="showExportDialog = true"
        class="mb-4"
      />
      
      <!-- レスポンシブレイアウト -->
      <v-row>
        <!-- サマリーカード -->
        <v-col cols="12" md="4">
          <SummaryCards
            :summary="reportSummary"
            :loading="isLoading"
          />
        </v-col>
        
        <!-- メインチャート -->
        <v-col cols="12" md="8">
          <v-card>
            <v-card-title>
              <v-chip-group v-model="selectedChartType" mandatory>
                <v-chip value="monthly">月次推移</v-chip>
                <v-chip value="category">カテゴリ別</v-chip>
                <v-chip value="weekday">曜日別</v-chip>
                <v-chip value="hourly">時間帯別</v-chip>
              </v-chip-group>
            </v-card-title>
            
            <v-card-text>
              <ReportCharts
                :chart-type="selectedChartType"
                :chart-data="chartData"
                :loading="isChartLoading"
              />
            </v-card-text>
          </v-card>
        </v-col>
      </v-row>
      
      <!-- カテゴリ別詳細分析 -->
      <CategoryAnalysisTable
        :analysis-data="categoryAnalysis"
        :loading="isLoading"
        class="mt-6"
      />
      
      <!-- 目標達成状況 -->
      <GoalProgressPanel
        :goal-progress="goalProgress"
        :loading="isLoading"
        class="mt-6"
      />
    </v-container>
    
    <!-- エクスポートダイアログ -->
    <ExportDialog
      v-model="showExportDialog"
      :filters="filters"
      @export="handleExport"
    />
  </div>
</template>

<script setup lang="ts">
import { ref, reactive, onMounted, watch, computed } from 'vue'
import { useReportStore } from '@/stores/report'
import { 
  ReportFilters as ReportFiltersType,
  ReportSummary,
  ChartData,
  CategoryAnalysis,
  GoalProgress
} from '@/types/report'

const reportStore = useReportStore()

const isLoading = ref(false)
const isChartLoading = ref(false)
const showExportDialog = ref(false)
const selectedChartType = ref('monthly')

const filters = reactive<ReportFiltersType>({
  startDate: new Date(new Date().getFullYear(), 0, 1), // 今年の開始
  endDate: new Date(),
  categories: [],
  comparisonPeriod: 'previous-year'
})

const reportSummary = ref<ReportSummary>({
  totalExpense: 0,
  averageDaily: 0,
  maxExpenseDate: null,
  maxExpenseAmount: 0,
  comparisonRate: 0
})

const chartData = ref<ChartData | null>(null)
const categoryAnalysis = ref<CategoryAnalysis[]>([])
const goalProgress = ref<GoalProgress | null>(null)

onMounted(() => {
  loadReportData()
})

watch([() => filters.startDate, () => filters.endDate], () => {
  loadReportData()
})

watch(selectedChartType, () => {
  loadChartData()
})

const loadReportData = async () => {
  try {
    isLoading.value = true
    
    const [summary, analysis, goals] = await Promise.all([
      reportStore.fetchReportSummary(filters),
      reportStore.fetchCategoryAnalysis(filters),
      reportStore.fetchGoalProgress(filters)
    ])
    
    reportSummary.value = summary
    categoryAnalysis.value = analysis
    goalProgress.value = goals
    
    await loadChartData()
    
  } catch (error) {
    console.error('Failed to load report data:', error)
  } finally {
    isLoading.value = false
  }
}

const loadChartData = async () => {
  try {
    isChartLoading.value = true
    chartData.value = await reportStore.fetchChartData(selectedChartType.value, filters)
  } catch (error) {
    console.error('Failed to load chart data:', error)
  } finally {
    isChartLoading.value = false
  }
}

const handleExport = async (exportOptions: any) => {
  try {
    await reportStore.exportReport({
      ...filters,
      ...exportOptions
    })
  } catch (error) {
    console.error('Failed to export report:', error)
  }
}
</script>
```

### MonthlyTrendChart.vue
```vue
<template>
  <div class="monthly-trend-chart">
    <canvas ref="chartCanvas" :height="height" />
  </div>
</template>

<script setup lang="ts">
import { ref, onMounted, watch, nextTick } from 'vue'
import Chart from 'chart.js/auto'
import { ChartData } from '@/types/report'

interface Props {
  data: ChartData | null
  height?: number
}

const props = withDefaults(defineProps<Props>(), {
  height: 400
})

const chartCanvas = ref<HTMLCanvasElement>()
let chartInstance: Chart | null = null

onMounted(() => {
  initChart()
})

watch(() => props.data, () => {
  updateChart()
}, { deep: true })

const initChart = () => {
  if (!chartCanvas.value || !props.data) return
  
  const ctx = chartCanvas.value.getContext('2d')
  if (!ctx) return
  
  chartInstance = new Chart(ctx, {
    type: 'line',
    data: {
      labels: props.data.labels,
      datasets: [
        {
          label: '今期',
          data: props.data.currentPeriod,
          borderColor: '#1976D2',
          backgroundColor: 'rgba(25, 118, 210, 0.1)',
          fill: true,
          tension: 0.4
        },
        {
          label: '前期',
          data: props.data.previousPeriod || [],
          borderColor: '#FFA726',
          backgroundColor: 'rgba(255, 167, 38, 0.1)',
          fill: false,
          borderDash: [5, 5]
        }
      ]
    },
    options: {
      responsive: true,
      maintainAspectRatio: false,
      interaction: {
        intersect: false,
        mode: 'index'
      },
      plugins: {
        title: {
          display: true,
          text: '月次支出推移'
        },
        legend: {
          position: 'top'
        },
        tooltip: {
          callbacks: {
            label: function(context) {
              return `${context.dataset.label}: ¥${context.parsed.y.toLocaleString()}`
            }
          }
        }
      },
      scales: {
        x: {
          display: true,
          title: {
            display: true,
            text: '月'
          }
        },
        y: {
          display: true,
          title: {
            display: true,
            text: '支出額 (円)'
          },
          ticks: {
            callback: function(value) {
              return '¥' + value.toLocaleString()
            }
          }
        }
      }
    }
  })
}

const updateChart = () => {
  if (!chartInstance || !props.data) return
  
  chartInstance.data.labels = props.data.labels
  chartInstance.data.datasets[0].data = props.data.currentPeriod
  chartInstance.data.datasets[1].data = props.data.previousPeriod || []
  
  chartInstance.update()
}
</script>
```

<script setup lang="ts">
import { computed } from 'vue'
import { CategoryAnalysis } from '@/types/report'

interface Props {
  analysisData: CategoryAnalysis[]
  loading?: boolean
}

const props = defineProps<Props>()

const headers = [
  { title: 'カテゴリ', key: 'category', sortable: false },
  { title: '今期', key: 'currentAmount', align: 'end' },
  { title: '前期', key: 'previousAmount', align: 'end' },
  { title: '増減', key: 'changeRate', align: 'center' },

const getChangeRateColor = (rate: number) => {
  if (rate > 10) return 'error'
  if (rate > 0) return 'warning'
  return 'success'

  switch (indicator) {
    case 'up': return 'error'
    case 'down': return 'success'
    default: return 'grey'
  }

    case 'up': return 'mdi-trending-up'

    default: return 'mdi-trending-neutral'
  }
}

## エクスポート機能

### データエクスポート
```vue
<template>
  <v-dialog v-model="showDialog" max-width="500">
    <v-card>
      <v-card-title>レポートエクスポート</v-card-title>
      
      <v-card-text>
        <v-form @submit.prevent="handleExport">
          <!-- ファイル形式 -->
          <v-radio-group v-model="exportOptions.format">
            <v-radio label="Excel (.xlsx)" value="xlsx" />
            <v-radio label="CSV" value="csv" />
            <v-radio label="PDF" value="pdf" />
          </v-radio-group>
          
          <!-- 含める内容 -->
          <v-checkbox
            v-model="exportOptions.includeCharts"
            label="グラフを含める"
          />
          <v-checkbox
            v-model="exportOptions.includeRawData"
            label="生データを含める"
          />
          <v-checkbox
            v-model="exportOptions.includeSummary"
            label="サマリーを含める"
          />
        </v-form>
      </v-card-text>
      
      <v-card-actions>
        <v-spacer />
        <v-btn @click="showDialog = false">キャンセル</v-btn>
        <v-btn @click="handleExport" color="primary" :loading="isExporting">
          エクスポート
        </v-btn>
      </v-card-actions>
    </v-card>
  </v-dialog>
</template>

<script setup lang="ts">
import { ref, reactive } from 'vue'

interface ExportOptions {
  format: 'xlsx' | 'csv' | 'pdf'
  includeCharts: boolean
  includeRawData: boolean
  includeSummary: boolean
}

const showDialog = ref(false)
const isExporting = ref(false)

const exportOptions = reactive<ExportOptions>({
  format: 'xlsx',
  includeCharts: true,
  includeRawData: false,
  includeSummary: true
})

const handleExport = async () => {
  try {
    isExporting.value = true
    
    const response = await fetch('/api/reports/export', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        filters: props.filters,
        options: exportOptions
      })
    })
    
    if (response.ok) {
      const blob = await response.blob()
      const url = window.URL.createObjectURL(blob)
      const link = document.createElement('a')
      
      link.href = url
      link.download = `report_${new Date().toISOString().split('T')[0]}.${exportOptions.format}`
      link.click()
      
      window.URL.revokeObjectURL(url)
      showDialog.value = false
    }
  } catch (error) {
    console.error('Export failed:', error)
  } finally {
    isExporting.value = false
  }
}
</script>
```
