# 支出記録画面設計

## 画面概要

新しい支出記録の入力・編集を行う画面です。カテゴリ選択、金額入力、日付選択などの機能を提供します。

### ExpenseRecordView.vue
```vue
<template>
  <div class="expense-record-view">
    <AppHeader title="支出記録" />
    
    <v-container class="pa-4">
      <v-row>
        <!-- メインフォーム -->
        <v-col cols="12" lg="8">
          <v-card>
            <v-card-title>
              <v-icon class="me-2">mdi-credit-card-plus</v-icon>
              {{ isEditing ? '支出記録編集' : '支出記録入力' }}
            </v-card-title>
            
            <v-card-text>
              <ExpenseForm
                :expense="currentExpense"
                :loading="isSubmitting"
                @submit="handleSubmit"
                @cancel="handleCancel"
              />
            </v-card-text>
          </v-card>
        </v-col>
        
        <!-- サイドバー -->
        <v-col cols="12" lg="4">
          <!-- よく使う支出テンプレート -->
          <ExpenseTemplates 
            class="mb-4"
            @template-selected="applyTemplate"
          />
          
          <!-- 最近の支出 -->
          <RecentExpensesList 
            :expenses="recentExpenses"
            @expense-clicked="editExpense"
          />
        </v-col>
      </v-row>
    </v-container>
  </div>
</template>

<script setup lang="ts">
import { ref, onMounted } from 'vue'
import { useRoute, useRouter } from 'vue-router'
import { useExpenseStore } from '@/stores/expense'
import { Expense, CreateExpenseRequest } from '@/types/expense'

const route = useRoute()
const router = useRouter()
const expenseStore = useExpenseStore()

const isEditing = ref(false)
const isSubmitting = ref(false)
const currentExpense = ref<Partial<Expense>>({})
const recentExpenses = ref<Expense[]>([])

onMounted(async () => {
  // 編集モードの場合は既存データを読み込み
  if (route.params.id) {
    isEditing.value = true
    await loadExpense(route.params.id as string)
  }
  
  // 最近の支出を読み込み
  await loadRecentExpenses()
})

const loadExpense = async (id: string) => {
  try {
    currentExpense.value = await expenseStore.fetchExpense(id)
  } catch (error) {
    console.error('Failed to load expense:', error)
    router.push('/expenses')
  }
}

const loadRecentExpenses = async () => {
  try {
    recentExpenses.value = await expenseStore.fetchRecentExpenses(10)
  } catch (error) {
    console.error('Failed to load recent expenses:', error)
  }
}

const handleSubmit = async (expenseData: CreateExpenseRequest) => {
  try {
    isSubmitting.value = true
    
    if (isEditing.value) {
      await expenseStore.updateExpense(currentExpense.value.id!, expenseData)
    } else {
      await expenseStore.createExpense(expenseData)
    }
    
    router.push('/expenses')
  } catch (error) {
    console.error('Failed to save expense:', error)
  } finally {
    isSubmitting.value = false
  }
}

const handleCancel = () => {
  router.back()
}

const applyTemplate = (template: ExpenseTemplate) => {
  currentExpense.value = {
    ...currentExpense.value,
    categoryId: template.categoryId,
    amount: template.amount,
    description: template.description
  }
}

const editExpense = (expense: Expense) => {
  router.push(`/expenses/edit/${expense.id}`)
}
</script>
```

### ExpenseForm.vue
```vue
<template>
  <v-form @submit.prevent="onSubmit" ref="formRef">
    <v-row>
      <!-- カテゴリ選択 -->
      <v-col cols="12" md="6">
        <CategorySelector
          v-model="form.categoryId"
          :rules="categoryRules"
          required
        />
      </v-col>
      
      <!-- 金額入力 -->
      <v-col cols="12" md="6">
        <AmountInput
          v-model="form.amount"
          :rules="amountRules"
          required
        />
      </v-col>
      
      <!-- 支出日 -->
      <v-col cols="12" md="6">
        <DatePicker
          v-model="form.expenseDate"
          :rules="dateRules"
          required
        />
      </v-col>
      
      <!-- 説明 -->
      <v-col cols="12">
        <DescriptionTextarea
          v-model="form.description"
          :rules="descriptionRules"
        />
      </v-col>
      
      <!-- レシート画像 -->
      <v-col cols="12">
        <ReceiptUpload
          v-model="form.receiptImages"
          :max-files="3"
        />
      </v-col>
      
      <!-- アクションボタン -->
      <v-col cols="12">
        <FormActions
          :loading="loading"
          :is-editing="isEditing"
          @submit="onSubmit"
          @cancel="$emit('cancel')"
        />
      </v-col>
    </v-row>
  </v-form>
</template>

<script setup lang="ts">
import { ref, reactive, watch } from 'vue'
import { Expense, CreateExpenseRequest } from '@/types/expense'

interface Props {
  expense?: Partial<Expense>
  loading?: boolean
}

interface Emits {
  (e: 'submit', data: CreateExpenseRequest): void
  (e: 'cancel'): void
}

const props = withDefaults(defineProps<Props>(), {
  loading: false
})

const emit = defineEmits<Emits>()

const formRef = ref()
const isEditing = computed(() => !!props.expense?.id)

const form = reactive<CreateExpenseRequest>({
  categoryId: '',
  amount: 0,
  description: '',
  expenseDate: new Date().toISOString().split('T')[0],
  receiptImages: []
})

// propsの変更を監視してフォームを更新
watch(() => props.expense, (newExpense) => {
  if (newExpense) {
    Object.assign(form, {
      categoryId: newExpense.categoryId || '',
      amount: newExpense.amount || 0,
      description: newExpense.description || '',
      expenseDate: newExpense.expenseDate || new Date().toISOString().split('T')[0],
      receiptImages: newExpense.receiptImages || []
    })
  }
}, { immediate: true, deep: true })

// バリデーションルール
const categoryRules = [
  (v: string) => !!v || 'カテゴリを選択してください'
]

const amountRules = [
  (v: number) => !!v || '金額を入力してください',
  (v: number) => v > 0 || '金額は1円以上で入力してください',
  (v: number) => v <= 10000000 || '金額は1000万円以下で入力してください'
]

const dateRules = [
  (v: string) => !!v || '支出日を選択してください',
  (v: string) => new Date(v) <= new Date() || '未来の日付は選択できません'
]

const descriptionRules = [
  (v: string) => !v || v.length <= 500 || '説明は500文字以内で入力してください'
]

const onSubmit = async () => {
  const { valid } = await formRef.value.validate()
  if (valid) {
    emit('submit', { ...form })
  }
}
</script>
```

### ReceiptUpload.vue
```vue
<template>
  <div class="receipt-upload">
    <v-file-input
      v-model="files"
      :rules="fileRules"
      label="レシート画像"
      prepend-icon="mdi-receipt"
      accept="image/*"
      multiple
      :max-files="maxFiles"
      show-size
      variant="outlined"
    >
      <template #selection="{ fileNames }">
        <template v-for="(fileName, index) in fileNames" :key="fileName">
          <v-chip
            :key="fileName"
            color="primary"
            size="small"
            label
            class="me-2"
          >
            {{ fileName }}
          </v-chip>
        </template>
      </template>
    </v-file-input>
    
    <!-- カメラ撮影ボタン -->
    <div class="d-flex gap-2 mt-2">
      <v-btn
        @click="openCamera"
        prepend-icon="mdi-camera"
        variant="outlined"
        size="small"
      >
        写真撮影
      </v-btn>
      
      <v-btn
        v-if="files.length > 0"
        @click="clearFiles"
        prepend-icon="mdi-delete"
        variant="outlined"
        color="error"
        size="small"
      >
        クリア
      </v-btn>
    </div>
    
    <!-- プレビュー -->
    <div v-if="files.length > 0" class="mt-4">
      <v-row>
        <v-col
          v-for="(file, index) in files"
          :key="index"
          cols="6"
          sm="4"
        >
          <v-card>
            <v-img
              :src="getFileUrl(file)"
              height="120"
              cover
            />
            <v-card-actions class="pa-2">
              <v-btn
                @click="removeFile(index)"
                icon="mdi-delete"
                size="small"
                color="error"
              />
            </v-card-actions>
          </v-card>
        </v-col>
      </v-row>
    </div>
    
    <!-- カメラダイアログ -->
    <v-dialog v-model="showCamera" max-width="500">
      <v-card>
        <v-card-title>写真撮影</v-card-title>
        <v-card-text>
          <video ref="videoRef" autoplay playsinline width="100%" />
          <canvas ref="canvasRef" style="display: none;" />
        </v-card-text>
        <v-card-actions>
          <v-spacer />
          <v-btn @click="closeCamera">キャンセル</v-btn>
          <v-btn @click="capturePhoto" color="primary">撮影</v-btn>
        </v-card-actions>
      </v-card>
    </v-dialog>
  </div>

<script setup lang="ts">
import { ref, computed } from 'vue'

interface Props {
  modelValue?: File[]
  maxFiles?: number
}

interface Emits {
  (e: 'update:modelValue', files: File[]): void
}

const props = withDefaults(defineProps<Props>(), {
  maxFiles: 3
})

const emit = defineEmits<Emits>()

const showCamera = ref(false)
const videoRef = ref<HTMLVideoElement>()
const canvasRef = ref<HTMLCanvasElement>()
const stream = ref<MediaStream>()

const files = computed({
  get: () => props.modelValue || [],
  set: (value: File[]) => emit('update:modelValue', value)
})

const fileRules = [
  (value: File[]) => !value || value.length <= props.maxFiles || `最大${props.maxFiles}ファイルまでです`,
  (value: File[]) => !value || value.every(file => file.size <= 5 * 1024 * 1024) || 'ファイルサイズは5MB以下にしてください'
]

const getFileUrl = (file: File) => URL.createObjectURL(file)

const removeFile = (index: number) => {
  const newFiles = [...files.value]
  newFiles.splice(index, 1)
  files.value = newFiles
}

const clearFiles = () => {
  files.value = []
}

const openCamera = async () => {
  try {
    stream.value = await navigator.mediaDevices.getUserMedia({ 
      video: { facingMode: 'environment' } // 背面カメラを優先
    })
    showCamera.value = true
    
    nextTick(() => {
      if (videoRef.value && stream.value) {
        videoRef.value.srcObject = stream.value
      }
    })
  } catch (error) {
    console.error('Camera access failed:', error)
    alert('カメラへのアクセスが許可されていません')
  }
}

const closeCamera = () => {
  if (stream.value) {
    stream.value.getTracks().forEach(track => track.stop())
  }
  showCamera.value = false
}

const capturePhoto = () => {
  if (!videoRef.value || !canvasRef.value) return
  
  const video = videoRef.value
  const canvas = canvasRef.value
  const context = canvas.getContext('2d')
  
  canvas.width = video.videoWidth
  canvas.height = video.videoHeight
  
  context?.drawImage(video, 0, 0)
  
  canvas.toBlob((blob) => {
    if (blob) {
      const file = new File([blob], `receipt-${Date.now()}.jpg`, { type: 'image/jpeg' })
      files.value = [...files.value, file]
    }
  }, 'image/jpeg', 0.8)
  
  closeCamera()
}
</script>
```

```vue
<template>
  <v-btn
    @click="startSpeechRecognition"
    :loading="isListening"
    icon="mdi-microphone"
    variant="outlined"
  />

<script setup>
const isListening = ref(false)

const startSpeechRecognition = () => {
  if (!('webkitSpeechRecognition' in window)) {
    alert('音声認識がサポートされていません')
    return
  }

  const recognition = new webkitSpeechRecognition()
  recognition.lang = 'ja-JP'
  recognition.continuous = false
  recognition.interimResults = false

  recognition.onstart = () => {
    isListening.value = true
  }

  recognition.onresult = (event) => {
    const result = event.results[0][0].transcript
    // 音声認識結果を説明欄に設定
    form.description = result
  }

  recognition.onend = () => {
    isListening.value = false
  }

  recognition.start()
}
</script>
```
