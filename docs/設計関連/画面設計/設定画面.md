# 設定画面設計

## 画面概要

### SettingsView.vue
```vue
<template>
  <div class="settings-view">
    <AppHeader title="設定" />
    
    <v-container class="pa-4">
      <!-- デスクトップレイアウト -->
      <v-row v-if="!isMobile">
        <v-col cols="3">
          <SettingsNavigation
            v-model="selectedSection"
            :sections="settingSections"
          />
        </v-col>
        
        <v-col cols="9">
          <SettingsContent :section="selectedSection" />
        </v-col>
      </v-row>
      
      <!-- モバイルレイアウト -->
      <div v-else>
        <SettingsList
          v-if="!selectedSection"
          :sections="settingSections"
          @select="handleSectionSelect"
        />
        
        <SettingsContent
          v-else
          :section="selectedSection"
          :show-back="true"
          @back="selectedSection = null"
        />
      </div>
    </v-container>
  </div>
</template>

<script setup lang="ts">
import { ref, computed, onMounted } from 'vue'
import { useDisplay } from 'vuetify'
import { useRouter, useRoute } from 'vue-router'

const { mobile } = useDisplay()
const router = useRouter()
const route = useRoute()

const isMobile = computed(() => mobile.value)
const selectedSection = ref<string | null>(null)

const settingSections = [
  {
    id: 'profile',
    title: 'ユーザー情報',
    icon: 'mdi-account',
    description: 'プロフィール・アカウント情報'
  },
  {
    id: 'categories',
    title: 'カテゴリ',
    icon: 'mdi-tag-multiple',
    description: 'カテゴリの追加・編集・削除'
  },
  {
    id: 'notifications',
    title: '通知',
    icon: 'mdi-bell',
    description: '通知設定・メール配信'
  },
  {
    id: 'security',
    title: 'セキュリティ',
    icon: 'mdi-security',
    description: '二段階認証・セッション管理'
  },
  {
    id: 'display',
    title: '表示',
    icon: 'mdi-palette',
    description: 'テーマ・表示設定'
  },
  {
    id: 'data',
    title: 'データ',
    icon: 'mdi-database',
    description: 'データの管理・バックアップ'
  },
  {
    id: 'app-info',
    title: 'アプリ情報',
    icon: 'mdi-information',
    description: 'バージョン・ライセンス情報'
  }
]

onMounted(() => {
  // URLパラメータから初期セクションを設定
  const section = route.query.section as string
  if (section && settingSections.find(s => s.id === section)) {
    selectedSection.value = section
  } else if (!isMobile.value) {
    selectedSection.value = 'profile'
  }
})

const handleSectionSelect = (sectionId: string) => {
  selectedSection.value = sectionId
  router.push({ query: { section: sectionId } })
}
</script>
```

<script setup lang="ts">
import { ref, reactive, onMounted } from 'vue'
import { useUserStore } from '@/stores/user'
import { User, UpdateUserRequest } from '@/types/user'

const userStore = useUserStore()

const formRef = ref()
const isSaving = ref(false)
const isDeleting = ref(false)
const showPasswordDialog = ref(false)
const showDeleteDialog = ref(false)
const deleteConfirmation = ref('')

const currentUser = ref<User>({} as User)

const profile = reactive<UpdateUserRequest>({
  firstName: '',
  lastName: '',
  email: '',
  phoneNumber: '',
  birthDate: '',
  gender: '',
  avatar: null
})

const genderOptions = [
  { title: '男性', value: 'male' },
  { title: '女性', value: 'female' },
  { title: 'その他', value: 'other' },
  { title: '回答しない', value: 'prefer_not_to_say' }
]

// バリデーションルール
const nameRules = [
  (v: string) => !!v || '名前を入力してください',
  (v: string) => v.length <= 50 || '名前は50文字以内で入力してください'
]

const emailRules = [
  (v: string) => !!v || 'メールアドレスを入力してください',
  (v: string) => {
    const pattern = /^[^\s@]+@[^\s@]+\.[^\s@]+$/
    return pattern.test(v) || '有効なメールアドレスを入力してください'
  }
]

const phoneRules = [
  (v: string) => !v || v.length <= 20 || '電話番号は20文字以内で入力してください',
  (v: string) => !v || /^[0-9-+().\s]+$/.test(v) || '有効な電話番号を入力してください'
]

onMounted(async () => {
  currentUser.value = await userStore.getCurrentUser()
  
  // フォームに現在の値を設定
  Object.assign(profile, {
    firstName: currentUser.value.firstName,
    lastName: currentUser.value.lastName,
    email: currentUser.value.email,
    phoneNumber: currentUser.value.phoneNumber,
    birthDate: currentUser.value.birthDate,
    gender: currentUser.value.gender
  })
})

const handleSubmit = async () => {
  const { valid } = await formRef.value.validate()
  if (!valid) return
  
  try {
    isSaving.value = true
    await userStore.updateProfile(profile)
    
    // 成功メッセージ
    // TODO: トースト通知を表示
    
  } catch (error) {
    console.error('Failed to update profile:', error)
  } finally {
    isSaving.value = false
  }
}

const resetForm = () => {
  Object.assign(profile, {
    firstName: currentUser.value.firstName,
    lastName: currentUser.value.lastName,
    email: currentUser.value.email,
    phoneNumber: currentUser.value.phoneNumber,
    birthDate: currentUser.value.birthDate,
    gender: currentUser.value.gender
  })
}

const handleAvatarUpload = async (file: File) => {
  try {
    const formData = new FormData()
    formData.append('avatar', file)
    
    await userStore.uploadAvatar(formData)
    
    // アバターを更新
    currentUser.value = await userStore.getCurrentUser()
    
  } catch (error) {
    console.error('Failed to upload avatar:', error)
  }
}

const handlePasswordChanged = () => {

  // TODO: パスワード変更成功メッセージ

const handleAccountDelete = async () => {

### CategorySettings.vue
```vue
<template>
  <div class="category-settings">
    <v-card>
      <v-card-title class="d-flex justify-space-between align-center">
        <span>
          <v-icon class="me-2">mdi-tag-multiple</v-icon>
          カテゴリ管理
        </span>
        <v-btn
          @click="showCreateDialog = true"
          color="primary"
          prepend-icon="mdi-plus"
        >
          新規作成
        </v-btn>
      </v-card-title>
      
      <v-card-text>
        <CategoryList
          :categories="categories"
          :loading="isLoading"
          @edit="handleEdit"
          @delete="handleDelete"
          @reorder="handleReorder"
        />
      </v-card-text>
    </v-card>
    
    <!-- 作成・編集ダイアログ -->
    <CategoryEditDialog
      v-model="showEditDialog"
      :category="selectedCategory"
      @saved="handleCategorySaved"
    />
    
    <!-- 削除確認ダイアログ -->
    <CategoryDeleteDialog
      v-model="showDeleteDialog"
      :category="selectedCategory"
      @deleted="handleCategoryDeleted"
    />
  </div>
</template>

<script setup lang="ts">
import { ref, onMounted } from 'vue'
import { useCategoryStore } from '@/stores/category'
import { Category } from '@/types/category'

const categoryStore = useCategoryStore()

const isLoading = ref(false)
const showCreateDialog = ref(false)
const showEditDialog = ref(false)
const showDeleteDialog = ref(false)
const selectedCategory = ref<Category | null>(null)
const categories = ref<Category[]>([])

// showCreateDialogの監視でshowEditDialogを制御
watch(showCreateDialog, (value) => {
  if (value) {
    selectedCategory.value = null
    showEditDialog.value = true
  }
})

onMounted(() => {
  loadCategories()
})

const loadCategories = async () => {
  try {
    isLoading.value = true
    categories.value = await categoryStore.fetchCategories()
  } catch (error) {
    console.error('Failed to load categories:', error)
  } finally {
    isLoading.value = false
  }
}

const handleEdit = (category: Category) => {
  selectedCategory.value = category
  showEditDialog.value = true
}

const handleDelete = (category: Category) => {
  selectedCategory.value = category
  showDeleteDialog.value = true
}

const handleReorder = async (reorderedCategories: Category[]) => {
  try {
    await categoryStore.reorderCategories(reorderedCategories)
    categories.value = reorderedCategories
  } catch (error) {
    console.error('Failed to reorder categories:', error)
  }
}

const handleCategorySaved = () => {
  showCreateDialog.value = false
  showEditDialog.value = false
  selectedCategory.value = null
  loadCategories()
}

const handleCategoryDeleted = () => {
  showDeleteDialog.value = false
  selectedCategory.value = null
  loadCategories()
}
</script>
```

const applyTheme = (themeName: string) => {
  if (themeName === 'system') {
    const prefersDark = window.matchMedia('(prefers-color-scheme: dark)').matches
    theme.global.name.value = prefersDark ? 'dark' : 'light'
  } else {
    theme.global.name.value = themeName
  }
}

const applyColorTheme = (colorName: string) => {
  const selectedColor = colorThemes.find(c => c.name === colorName)
  if (selectedColor) {
    theme.themes.value.light.colors.primary = selectedColor.primary
    theme.themes.value.dark.colors.primary = selectedColor.primary
  }
}

  } finally {
    isSaving.value = false
