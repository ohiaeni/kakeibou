# 予算設定画面設計

## 画面概要

月次予算の設定・編集・管理を行う画面です。カテゴリごとの予算設定と予算達成状況の確認が可能です。

```vue
      :year="selectedYear"
      :month="selectedMonth"
      @saved="handleBudgetSaved"
    />
    
    <!-- モバイル編集シート -->
    <BudgetEditSheet
      v-model="showEditSheet"
      :budget="selectedBudget"
      :year="selectedYear"
      :month="selectedMonth"
      @saved="handleBudgetSaved"
    />
    
    <!-- 予算コピーダイアログ -->
    <BudgetTemplateDialog
      v-model="showCopyDialog"
      :target-year="selectedYear"
      :target-month="selectedMonth"
      @copied="loadBudgetData"
    />

<script setup lang="ts">
import { ref, onMounted, computed, watch } from 'vue'
import { useDisplay } from 'vuetify'
import { useBudgetStore } from '@/stores/budget'
import { Budget, BudgetSummary, BudgetAnalytics } from '@/types/budget'

const { mobile } = useDisplay()
const budgetStore = useBudgetStore()

const isMobile = computed(() => mobile.value)
const isLoading = ref(false)
const isAnalyticsLoading = ref(false)

const selectedYear = ref(new Date().getFullYear())
const selectedMonth = ref(new Date().getMonth() + 1)

const budgets = ref<Budget[]>([])
const budgetSummary = ref<BudgetSummary>({
  totalBudget: 0,
  totalSpent: 0,
  totalRemaining: 0,
  achievementRate: 0
})
const analyticsData = ref<BudgetAnalytics | null>(null)

const showEditDialog = ref(false)
const showEditSheet = ref(false)
const showCopyDialog = ref(false)
const selectedBudget = ref<Budget | null>(null)

onMounted(() => {
  loadBudgetData()
  loadAnalyticsData()
})

watch([selectedYear, selectedMonth], () => {
  loadBudgetData()
})

const loadBudgetData = async () => {
  try {
    isLoading.value = true
    
    const result = await budgetStore.fetchBudgets(selectedYear.value, selectedMonth.value)
    budgets.value = result.budgets
    budgetSummary.value = result.summary
    
  } catch (error) {
    console.error('Failed to load budget data:', error)
  } finally {
    isLoading.value = false
  }
}

const loadAnalyticsData = async () => {
  try {
    isAnalyticsLoading.value = true
    analyticsData.value = await budgetStore.fetchAnalytics(selectedYear.value)
  } catch (error) {
    console.error('Failed to load analytics data:', error)
  } finally {
    isAnalyticsLoading.value = false
  }
}

const openEditDialog = (budget: Budget) => {
  selectedBudget.value = budget
  showEditDialog.value = true
}

const openEditSheet = (budget: Budget) => {
  selectedBudget.value = budget
  showEditSheet.value = true
}

const addNewCategory = () => {
  selectedBudget.value = null
  if (isMobile.value) {
    showEditSheet.value = true
  } else {
    showEditDialog.value = true
  }
}

const handleBudgetSaved = () => {
  showEditDialog.value = false
  showEditSheet.value = false
  selectedBudget.value = null
  loadBudgetData()
}

const confirmDelete = async (budget: Budget) => {
  if (confirm(`${budget.category.name}の予算を削除しますか？`)) {
    try {
      await budgetStore.deleteBudget(budget.id)
      await loadBudgetData()
    } catch (error) {
      console.error('Failed to delete budget:', error)
    }
  }
}
</script>
```

```vue
<script setup lang="ts">
import { computed } from 'vue'
import { BudgetSummary } from '@/types/budget'

interface Props {
  summary: BudgetSummary
  loading?: boolean
}

const props = defineProps<Props>()

const remainingColorClass = computed(() => {
  if (props.summary.totalRemaining < 0) return 'text-error'
  if (props.summary.totalRemaining < props.summary.totalBudget * 0.1) return 'text-warning'

const progressColor = computed(() => {
  const rate = props.summary.achievementRate

  return 'success'
```

### BudgetEditDialog.vue
```vue
<template>
  <v-dialog
    :model-value="modelValue"
    @update:model-value="$emit('update:modelValue', $event)"
    max-width="500"
    persistent
  >
    <v-card>
      <v-card-title>
        {{ isEditing ? '予算編集' : '新規予算設定' }}
      </v-card-title>
      
      <v-card-text>
        <v-form @submit.prevent="handleSubmit" ref="formRef">
          <!-- カテゴリ選択 -->
          <v-select
            v-model="form.categoryId"
            :items="categories"
            :rules="categoryRules"
            label="カテゴリ"
            item-title="name"
            item-value="id"
            variant="outlined"
            class="mb-3"
            :disabled="isEditing"
          >
            <template #item="{ props, item }">
              <v-list-item v-bind="props">
                <template #prepend>
                  <v-icon :color="item.raw.color">{{ item.raw.icon }}</v-icon>
                </template>
              </v-list-item>
            </template>
          </v-select>
          
          <!-- 予算金額 -->
          <v-text-field
            v-model.number="form.amount"
            :rules="amountRules"
            label="予算金額"
            type="number"
            prefix="¥"
            variant="outlined"
            class="mb-3"
          />
          
          <!-- 予算タイプ -->
          <v-radio-group v-model="form.budgetType" inline class="mb-3">
            <v-radio label="固定予算" value="fixed" />
            <v-radio label="変動予算" value="flexible" />
          </v-radio-group>
          
          <!-- メモ -->
          <v-textarea
            v-model="form.memo"
            label="メモ"
            variant="outlined"
            rows="3"
            :rules="memoRules"
          />
        </v-form>
      </v-card-text>
      
      <v-card-actions>
        <v-spacer />
        <v-btn @click="$emit('update:modelValue', false)">
          キャンセル
        </v-btn>
        <v-btn
          @click="handleSubmit"
          color="primary"
          :loading="isSubmitting"
        >
          {{ isEditing ? '更新' : '作成' }}
        </v-btn>
      </v-card-actions>
    </v-card>
  </v-dialog>
</template>

<script setup lang="ts">
import { ref, reactive, computed, watch, onMounted } from 'vue'
import { useBudgetStore } from '@/stores/budget'
import { useCategoryStore } from '@/stores/category'
import { Budget, CreateBudgetRequest } from '@/types/budget'

interface Props {
  modelValue: boolean
  budget?: Budget | null
  year: number
  month: number
}

interface Emits {
  (e: 'update:modelValue', value: boolean): void
  (e: 'saved'): void
}

const props = defineProps<Props>()
const emit = defineEmits<Emits>()

const budgetStore = useBudgetStore()
const categoryStore = useCategoryStore()

const formRef = ref()
const isSubmitting = ref(false)
const categories = ref([])

const isEditing = computed(() => !!props.budget?.id)

const form = reactive<CreateBudgetRequest>({
  categoryId: '',
  amount: 0,
  budgetType: 'fixed',
  memo: ''
})

// バリデーションルール
const categoryRules = [
  (v: string) => !!v || 'カテゴリを選択してください'
]

const amountRules = [
  (v: number) => !!v || '予算金額を入力してください',
  (v: number) => v > 0 || '予算金額は1円以上で入力してください',
  (v: number) => v <= 10000000 || '予算金額は1000万円以下で入力してください'
]

const memoRules = [
  (v: string) => !v || v.length <= 200 || 'メモは200文字以内で入力してください'
]

onMounted(async () => {
  categories.value = await categoryStore.fetchCategories()
})

watch(() => props.budget, (newBudget) => {
  if (newBudget) {
    form.categoryId = newBudget.categoryId
    form.amount = newBudget.amount
    form.budgetType = newBudget.budgetType || 'fixed'
    form.memo = newBudget.memo || ''
  } else {
    // 新規作成時はフォームをリセット
    Object.assign(form, {
      categoryId: '',
      amount: 0,
      budgetType: 'fixed',
      memo: ''
    })
  }
}, { immediate: true })

const handleSubmit = async () => {
  const { valid } = await formRef.value.validate()
  if (!valid) return
  
  try {
    isSubmitting.value = true
    
    if (isEditing.value) {
      await budgetStore.updateBudget(props.budget!.id, {
        ...form,
        year: props.year,
        month: props.month
      })
    } else {
      await budgetStore.createBudget({
        ...form,
        year: props.year,
        month: props.month
      })
    }
    
    emit('saved')
  } catch (error) {
    console.error('Failed to save budget:', error)
  } finally {
    isSubmitting.value = false
  }
}
</script>
```

## 機能仕様

### 予算コピー機能
```typescript
// stores/budget.ts
export const useBudgetStore = defineStore('budget', {
  actions: {
    async copyBudgetFromMonth(
      sourceYear: number,
      sourceMonth: number,
      targetYear: number,
      targetMonth: number
    ) {
      try {
        const sourceBudgets = await this.fetchBudgets(sourceYear, sourceMonth)
        
        const copyPromises = sourceBudgets.budgets.map(budget => 
          this.createBudget({
            categoryId: budget.categoryId,
            amount: budget.amount,
            budgetType: budget.budgetType,
            memo: budget.memo,
            year: targetYear,
            month: targetMonth
          })
        )
        
        await Promise.all(copyPromises)
      } catch (error) {
        throw new Error('予算のコピーに失敗しました')
      }
    }
  }
})
```

### 自動予算調整

```typescript
// 過去の支出実績に基づく予算提案
const suggestBudgetAmount = async (categoryId: string) => {
  const lastThreeMonths = await expenseStore.getCategoryExpenses(categoryId, 3)
  const average = lastThreeMonths.reduce((sum, expense) => sum + expense.amount, 0) / 3
  
  // 10%のバッファを追加
  return Math.ceil(average * 1.1)
}
```
