### 2. クエリ最適化

#### N+1問題の回避
```java
@Repository
public class ExpenseRepository {
    
    // 悪い例: N+1問題が発生
    public List<ExpenseEntity> findExpensesWithCategories_Bad(UUID userId) {
        List<ExpenseEntity> expenses = findByUserId(userId);
        expenses.forEach(expense -> {
            expense.setCategory(categoryRepository.findById(expense.getCategoryId()));
        });
        return expenses;
    }
    
    // 良い例: JOINを使用して一度に取得
    @Select("""
        SELECT e.*, c.name as category_name, c.description as category_description
        FROM expense_records e
        LEFT JOIN expense_categories c ON e.category_id = c.id
        WHERE e.user_id = #{userId}
        ORDER BY e.expense_date DESC
        """)
    List<ExpenseWithCategoryEntity> findExpensesWithCategories_Good(@Param("userId") UUID userId);
}
```

#### ページネーション最適化
```java
@Service
public class ExpenseService {
    
    public PagedResult<ExpenseDto> getExpenses(UUID userId, int page, int size) {
        // OFFSET/LIMITよりもカーソルベースページネーションを推奨
        LocalDate cursor = getCursorFromPage(page);
        
        List<ExpenseEntity> expenses = expenseRepository.findExpensesAfterDate(
            userId, cursor, size + 1
        );
        
        boolean hasNext = expenses.size() > size;
        if (hasNext) {
            expenses = expenses.subList(0, size);
        }
        
        return PagedResult.<ExpenseDto>builder()
            .items(expenses.stream().map(this::toDto).toList())
            .hasNext(hasNext)
            .nextCursor(hasNext ? expenses.get(size - 1).getExpenseDate() : null)
            .build();
    }
}
```

## アプリケーションパフォーマンス

### 1. キャッシュ戦略

#### Spring Cache設定
```java
@Configuration
@EnableCaching
public class CacheConfig {
    
    @Bean
    public CacheManager cacheManager() {
        RedisCacheManager.Builder builder = RedisCacheManager
            .RedisCacheManagerBuilder
            .fromConnectionFactory(redisConnectionFactory())
            .cacheDefaults(cacheConfiguration());
        
        return builder.build();
    }
    
    private RedisCacheConfiguration cacheConfiguration() {
        return RedisCacheConfiguration.defaultCacheConfig()
            .entryTtl(Duration.ofMinutes(30))
            .serializeKeysWith(RedisSerializationContext.SerializationPair
                .fromSerializer(new StringRedisSerializer()))
            .serializeValuesWith(RedisSerializationContext.SerializationPair
                .fromSerializer(new GenericJackson2JsonRedisSerializer()));
    }
}
```

#### キャッシュ適用例

@Service
public class CategoryService {
    
    @Cacheable(value = "categories", key = "'all'")
    public List<CategoryDto> getAllCategories() {
        // カテゴリマスターデータは変更頻度が低いためキャッシュ
        return categoryRepository.findAll()
            .stream()
            .map(this::toDto)
            .toList();
    }
    
    @Cacheable(value = "monthly-summary", key = "#userId + '_' + #year + '_' + #month")
    public MonthlySummaryDto getMonthlySummary(UUID userId, int year, int month) {
        // 月次集計結果をキャッシュ（過去データは変更されない）

    }
    
    @CacheEvict(value = "monthly-summary", key = "#userId + '_' + #year + '_' + #month")
    public void evictMonthlySummaryCache(UUID userId, int year, int month) {
        // 新しい支出記録追加時にキャッシュを無効化
    }
}

### 2. 非同期処理

#### 重い処理の非同期化
```java
@Service
public class ReportService {
    
    @Async("reportTaskExecutor")
    @Retryable(value = {Exception.class}, maxAttempts = 3)
    public CompletableFuture<String> generateAnnualReport(UUID userId, int year) {
        try {
            // 年次レポート生成（時間のかかる処理）
            String reportData = generateReportData(userId, year);
            
            // S3等にファイル保存
            String fileUrl = fileStorageService.saveReport(reportData);
            
            // ユーザーに通知
            notificationService.notifyReportReady(userId, fileUrl);
            
            return CompletableFuture.completedFuture(fileUrl);
        } catch (Exception e) {
            log.error("Failed to generate annual report for user: {}", userId, e);
            throw e;
        }
    }
}

@Configuration
@EnableAsync
public class AsyncConfig {
    
    @Bean("reportTaskExecutor")
    public TaskExecutor reportTaskExecutor() {
        ThreadPoolTaskExecutor executor = new ThreadPoolTaskExecutor();
        executor.setCorePoolSize(2);
        executor.setMaxPoolSize(5);
        executor.setQueueCapacity(25);
        executor.setThreadNamePrefix("ReportGen-");
        executor.initialize();
        return executor;
    }
}
```

@Component
public class DatabaseMetrics {
    
    private final MeterRegistry meterRegistry;
    private final DataSource dataSource;
    
    @EventListener
    @Async
    public void onExpenseCreated(ExpenseCreatedEvent event) {
        // 支出記録作成イベントをカウント
        meterRegistry.counter("business.expense.created",
            "category", event.getCategoryName())
            .increment();
    }
    
    @Scheduled(fixedRate = 60000) // 1分間隔
    public void recordDatabaseConnectionMetrics() {
        if (dataSource instanceof HikariDataSource hikariDataSource) {
            HikariPoolMXBean poolBean = hikariDataSource.getHikariPoolMXBean();
            
            Gauge.builder("database.connections.active")
                .register(meterRegistry, poolBean, HikariPoolMXBean::getActiveConnections);
            
            Gauge.builder("database.connections.idle")
                .register(meterRegistry, poolBean, HikariPoolMXBean::getIdleConnections);
        }
    }
}

import io.gatling.core.Predef._
import io.gatling.http.Predef._
import scala.concurrent.duration._

class ExpenseApiLoadTest extends Simulation {

  val httpProtocol = http
    .baseUrl("http://localhost:8080")
    .acceptHeader("application/json")
    .userAgentHeader("Gatling Load Test")

  val scn = scenario("Expense API Load Test")
    .exec(http("Login")
      .post("/api/v1/auth/login")
      .body(StringBody("""{"email":"test@example.com","password":"password"}"""))
      .check(jsonPath("$.access_token").saveAs("token")))
    .pause(1)
    .exec(http("Get Expenses")
      .get("/api/v1/expenses")
      .header("Authorization", "Bearer ${token}")
      .check(status.is(200)))
    .pause(2)
    .exec(http("Create Expense")
      .post("/api/v1/expenses")
      .header("Authorization", "Bearer ${token}")
      .body(StringBody("""{"categoryId":"uuid","amount":1500,"description":"test"}"""))
      .check(status.is(201)))

  setUp(
    scn.inject(
      rampUsers(100).during(5.minutes),
      constantUsersPerSec(20).during(10.minutes)
    )
  ).protocols(httpProtocol)
   .assertions(
     global.responseTime.percentile3.lt(1000),
     global.successfulRequests.percent.gt(95)
   )
}
