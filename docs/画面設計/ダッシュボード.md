### DashboardView.vue
```vue
<template>
  <div class="dashboard-view">
    <AppHeader />
    
    <v-container fluid class="pa-4">
      <!-- サマリーカード -->
      <DashboardSummary 
        :summary="dashboardData.summary" 
        :loading="isLoading"
      />
      
      <!-- クイックアクション -->
      <v-row class="mb-6">
        <v-col cols="12">
          <QuickActions @add-expense="openExpenseDialog" />
        </v-col>
      </v-row>
      
      <!-- チャートセクション -->
      <ChartsSection 
        :monthly-data="dashboardData.monthlyChart"
        :category-data="dashboardData.categoryChart"
        :loading="isLoading"
      />
      
      <!-- 最近の支出 -->
      <RecentExpensesSection 
        :expenses="dashboardData.recentExpenses"
        :loading="isLoading"
        @view-all="$router.push('/expenses')"
      />
    </v-container>
    
    <!-- 支出登録ダイアログ -->
    <ExpenseDialog 
      v-model="showExpenseDialog"
      @saved="handleExpenseSaved"
    />
  </div>
</template>

<script setup lang="ts">
import { ref, onMounted } from 'vue'
import { useDashboardStore } from '@/stores/dashboard'
import { DashboardData } from '@/types/dashboard'

const dashboardStore = useDashboardStore()
const isLoading = ref(false)
const showExpenseDialog = ref(false)
const dashboardData = ref<DashboardData>({
  summary: {
    currentMonthExpense: 0,
    currentMonthBudget: 0,
    remainingBudget: 0,
    expenseCount: 0
  },
  monthlyChart: [],
  categoryChart: [],
  recentExpenses: []
})

onMounted(async () => {
  await loadDashboardData()
})

const loadDashboardData = async () => {
  try {
    isLoading.value = true
    dashboardData.value = await dashboardStore.fetchDashboardData()
  } catch (error) {
    console.error('Failed to load dashboard data:', error)
  } finally {
    isLoading.value = false
  }
}

const openExpenseDialog = () => {
  showExpenseDialog.value = true
}

const handleExpenseSaved = () => {
  showExpenseDialog.value = false
  loadDashboardData() // データを再読み込み
}
</script>
```

### MonthlyExpenseChart.vue
```vue
<template>
  <v-card>
    <v-card-title>月次支出推移</v-card-title>
    <v-card-text>
      <canvas ref="chartCanvas" />
    </v-card-text>
  </v-card>
</template>

<script setup lang="ts">
import { ref, onMounted, watch } from 'vue'
import Chart from 'chart.js/auto'
import { MonthlyChartData } from '@/types/dashboard'

interface Props {
  data: MonthlyChartData[]
  loading?: boolean
}

const props = defineProps<Props>()
const chartCanvas = ref<HTMLCanvasElement>()
let chart: Chart | null = null

onMounted(() => {
  if (props.data.length > 0) {
    initChart()
  }
})

watch(() => props.data, (newData) => {
  if (newData.length > 0) {
    if (chart) {
      updateChart(newData)
    } else {
      initChart()
    }
  }
}, { deep: true })

const initChart = () => {
  if (!chartCanvas.value) return

  const ctx = chartCanvas.value.getContext('2d')
  if (!ctx) return

  chart = new Chart(ctx, {
    type: 'line',
    data: {
      labels: props.data.map(item => item.month),
      datasets: [
        {
          label: '支出',
          data: props.data.map(item => item.expense),
          borderColor: '#1976D2',
          backgroundColor: 'rgba(25, 118, 210, 0.1)',
          fill: true,
          tension: 0.4
        },
        {
          label: '予算',
          data: props.data.map(item => item.budget),
          borderColor: '#4CAF50',
          backgroundColor: 'transparent',
          borderDash: [5, 5]
        }
      ]
    },
    options: {
      responsive: true,
      maintainAspectRatio: false,
      plugins: {
        legend: {
          position: 'top'
        }
      },
      scales: {
        y: {
          beginAtZero: true,
          ticks: {
            callback: function(value) {
              return '¥' + Number(value).toLocaleString()
            }
          }
        }
      }
    }
  })
}

const updateChart = (data: MonthlyChartData[]) => {
  if (!chart) return

  chart.data.labels = data.map(item => item.month)
  chart.data.datasets[0].data = data.map(item => item.expense)
  chart.data.datasets[1].data = data.map(item => item.budget)
  chart.update()
}
</script>
```

## 状態管理（Pinia Store）

### dashboard.ts
```typescript
import { defineStore } from 'pinia'
import { dashboardApi } from '@/api/dashboard'
import { DashboardData } from '@/types/dashboard'

export const useDashboardStore = defineStore('dashboard', {
  state: () => ({
    dashboardData: null as DashboardData | null,
    lastUpdated: null as Date | null
  }),

  getters: {
    isDataStale: (state) => {
      if (!state.lastUpdated) return true
      const now = new Date()
      const diffMinutes = (now.getTime() - state.lastUpdated.getTime()) / (1000 * 60)
      return diffMinutes > 15 // 15分経過で古いデータとみなす
    }
  },

  actions: {
    async fetchDashboardData(): Promise<DashboardData> {
      try {
        // キャッシュされたデータが新しい場合はそれを返す
        if (this.dashboardData && !this.isDataStale) {
          return this.dashboardData
        }

        const data = await dashboardApi.getDashboardData()
        this.dashboardData = data
        this.lastUpdated = new Date()
        
        return data
      } catch (error) {
        console.error('Failed to fetch dashboard data:', error)
        throw error
      }
    },

    async refreshData() {
      this.lastUpdated = null // キャッシュを無効化
      return this.fetchDashboardData()
    }
  }
})
```

## リアルタイム更新

### WebSocket接続
```typescript
// composables/useRealtimeUpdates.ts
import { ref, onMounted, onUnmounted } from 'vue'
import { useAuthStore } from '@/stores/auth'

export const useRealtimeUpdates = () => {
  const authStore = useAuthStore()
  const isConnected = ref(false)
  let ws: WebSocket | null = null

  const connect = () => {
    if (!authStore.token) return

    ws = new WebSocket(`ws://localhost:8080/ws?token=${authStore.token}`)
    
    ws.onopen = () => {
      isConnected.value = true
      console.log('WebSocket connected')
    }
    
    ws.onmessage = (event) => {
      const data = JSON.parse(event.data)
      handleRealtimeUpdate(data)
    }
    
    ws.onclose = () => {
      isConnected.value = false
      console.log('WebSocket disconnected')
      // 自動再接続
      setTimeout(connect, 5000)
    }
  }

  const handleRealtimeUpdate = (data: any) => {
    if (data.type === 'expense_created') {
      // ダッシュボードデータを更新
      useDashboardStore().refreshData()
    }
  }

  onMounted(connect)
  onUnmounted(() => {
    if (ws) {
      ws.close()
    }
  })

  return {
    isConnected
  }
}
```
