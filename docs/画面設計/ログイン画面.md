### LoginView.vue
```vue
<template>
  <div class="login-view">
    <AppHeader title="家計簿アプリ" />
    
    <v-container class="fill-height">
      <v-row justify="center" align="center">
        <v-col cols="12" sm="8" md="6" lg="4">
          <v-card class="pa-6">
            <v-card-title class="text-center mb-4">
              <h2>ログイン</h2>
            </v-card-title>
            
            <v-card-text>
              <LoginForm @login="handleLogin" :loading="isLoading" />
            </v-card-text>
            
            <v-card-actions class="justify-center">
              <SignupLink />
            </v-card-actions>
          </v-card>
        </v-col>
      </v-row>
    </v-container>
  </div>
</template>

<script setup lang="ts">
import { ref } from 'vue'
import { useRouter } from 'vue-router'
import { useAuthStore } from '@/stores/auth'
import { LoginRequest } from '@/types/auth'

const router = useRouter()
const authStore = useAuthStore()
const isLoading = ref(false)

const handleLogin = async (credentials: LoginRequest) => {
  try {
    isLoading.value = true
    await authStore.login(credentials)
    router.push('/dashboard')
  } catch (error) {
    console.error('Login failed:', error)
  } finally {
    isLoading.value = false
  }
}
</script>
```

### LoginForm.vue
```vue
<template>
  <v-form @submit.prevent="onSubmit" ref="formRef">
    <v-text-field
      v-model="form.email"
      label="メールアドレス"
      type="email"
      :rules="emailRules"
      required
      prepend-inner-icon="mdi-email"
      variant="outlined"
      class="mb-3"
    />
    
    <v-text-field
      v-model="form.password"
      label="パスワード"
      :type="showPassword ? 'text' : 'password'"
      :rules="passwordRules"
      required
      prepend-inner-icon="mdi-lock"
      :append-inner-icon="showPassword ? 'mdi-eye' : 'mdi-eye-off'"
      @click:append-inner="showPassword = !showPassword"
      variant="outlined"
      class="mb-4"
    />
    
    <v-btn
      type="submit"
      color="primary"
      size="large"
      block
      :loading="loading"
      :disabled="!isFormValid"
    >
      ログイン
    </v-btn>
  </v-form>
</template>

<script setup lang="ts">
import { ref, computed } from 'vue'
import { LoginRequest } from '@/types/auth'

interface Props {
  loading?: boolean
}

interface Emits {
  (e: 'login', credentials: LoginRequest): void
}

const props = withDefaults(defineProps<Props>(), {
  loading: false
})

const emit = defineEmits<Emits>()

const formRef = ref()
const showPassword = ref(false)

const form = ref<LoginRequest>({
  email: '',
  password: ''
})

const emailRules = [
  (v: string) => !!v || 'メールアドレスは必須です',
  (v: string) => /.+@.+\..+/.test(v) || '有効なメールアドレスを入力してください'
]

const passwordRules = [
  (v: string) => !!v || 'パスワードは必須です',
  (v: string) => v.length >= 8 || 'パスワードは8文字以上で入力してください'
]

const isFormValid = computed(() => {
  return form.value.email && form.value.password && 
         emailRules.every(rule => rule(form.value.email) === true) &&
         passwordRules.every(rule => rule(form.value.password) === true)
})

const onSubmit = async () => {
  const { valid } = await formRef.value.validate()
  if (valid) {
    emit('login', form.value)
  }
}
</script>
```

## 状態管理（Pinia Store）

### auth.ts
```typescript
import { defineStore } from 'pinia'
import { ref } from 'vue'
import { authApi } from '@/api/auth'
import { LoginRequest, LoginResponse, User } from '@/types/auth'

export const useAuthStore = defineStore('auth', () => {
  const user = ref<User | null>(null)
  const token = ref<string | null>(localStorage.getItem('token'))
  const isAuthenticated = computed(() => !!token.value)

  const login = async (credentials: LoginRequest): Promise<void> => {
    try {
      const response: LoginResponse = await authApi.login(credentials)
      
      token.value = response.access_token
      user.value = response.user
      
      localStorage.setItem('token', response.access_token)
      localStorage.setItem('user', JSON.stringify(response.user))
      
    } catch (error) {
      throw new Error('ログインに失敗しました')
    }
  }

  const logout = (): void => {
    token.value = null
    user.value = null
    localStorage.removeItem('token')
    localStorage.removeItem('user')
  }

  return {
    user,
    token,
    isAuthenticated,
    login,
    logout
  }
})
```
