### ExpenseHistoryView.vue
```vue
<template>
  <div class="expense-history-view">
    <AppHeader title="支出履歴" />
    
    <v-container class="pa-4">
      <!-- 検索・フィルター -->
      <ExpenseSearchForm
        v-model="searchCriteria"
        @search="handleSearch"
        @clear="clearSearch"
        class="mb-4"
      />
      
      <!-- 集計情報 -->
      <ExpenseSummary
        :summary="expenseSummary"
        :loading="isLoading"
        class="mb-4"
      />
      
      <!-- 一覧表示 -->
      <v-card>
        <v-card-title class="d-flex justify-space-between align-center">
          <span>支出一覧</span>
          <div class="d-flex gap-2">
            <SortOptions v-model="sortOrder" />
            <ExportButton @export="exportExpenses" />
          </div>
        </v-card-title>
        
        <v-card-text class="pa-0">
          <!-- デスクトップ表示 -->
          <ExpenseTable
            v-if="!isMobile"
            :expenses="expenses"
            :loading="isLoading"
            @edit="editExpense"
            @delete="confirmDelete"
          />
          
          <!-- モバイル表示 -->
          <ExpenseCardList
            v-else
            :expenses="expenses"
            :loading="isLoading"
            @edit="editExpense"
            @delete="confirmDelete"
            @load-more="loadMoreExpenses"
          />
        </v-card-text>
      </v-card>
      
      <!-- ページネーション（デスクトップのみ） -->
      <PaginationControls
        v-if="!isMobile"
        :current-page="currentPage"
        :total-pages="totalPages"
        :total-items="totalItems"
        @page-change="handlePageChange"
        class="mt-4"
      />
    </v-container>
    
    <!-- 削除確認ダイアログ -->
    <ExpenseDeleteDialog
      v-model="showDeleteDialog"
      :expense="selectedExpense"
      @confirm="handleDelete"
    />
  </div>
</template>

<script setup lang="ts">
import { ref, onMounted, computed, watch } from 'vue'
import { useRouter } from 'vue-router'
import { useDisplay } from 'vuetify'
import { useExpenseStore } from '@/stores/expense'
import { Expense, ExpenseSearchCriteria, ExpenseSummary } from '@/types/expense'

const router = useRouter()
const { mobile } = useDisplay()
const expenseStore = useExpenseStore()

const isMobile = computed(() => mobile.value)
const isLoading = ref(false)
const expenses = ref<Expense[]>([])
const currentPage = ref(1)
const totalPages = ref(0)
const totalItems = ref(0)
const showDeleteDialog = ref(false)
const selectedExpense = ref<Expense | null>(null)

const searchCriteria = ref<ExpenseSearchCriteria>({
  keyword: '',
  categoryId: '',
  startDate: '',
  endDate: '',
  minAmount: null,
  maxAmount: null
})

const sortOrder = ref('date_desc')

const expenseSummary = ref<ExpenseSummary>({
  totalAmount: 0,
  totalCount: 0,
  averageAmount: 0
})

onMounted(() => {
  loadExpenses()
})

watch([searchCriteria, sortOrder], () => {
  currentPage.value = 1
  loadExpenses()
}, { deep: true })

const loadExpenses = async () => {
  try {
    isLoading.value = true
    
    const result = await expenseStore.searchExpenses({
      ...searchCriteria.value,
      page: currentPage.value,
      limit: isMobile.value ? 20 : 50,
      sortBy: sortOrder.value
    })
    
    expenses.value = result.expenses
    totalPages.value = result.totalPages
    totalItems.value = result.totalItems
    expenseSummary.value = result.summary
    
  } catch (error) {
    console.error('Failed to load expenses:', error)
  } finally {
    isLoading.value = false
  }
}

const loadMoreExpenses = async () => {
  if (currentPage.value >= totalPages.value) return
  
  try {
    currentPage.value++
    
    const result = await expenseStore.searchExpenses({
      ...searchCriteria.value,
      page: currentPage.value,
      limit: 20,
      sortBy: sortOrder.value
    })
    
    expenses.value.push(...result.expenses)
    
  } catch (error) {
    console.error('Failed to load more expenses:', error)
    currentPage.value-- // ロールバック
  }
}

const handleSearch = () => {
  currentPage.value = 1
  loadExpenses()
}

const clearSearch = () => {
  searchCriteria.value = {
    keyword: '',
    categoryId: '',
    startDate: '',
    endDate: '',
    minAmount: null,
    maxAmount: null
  }
}

const handlePageChange = (page: number) => {
  currentPage.value = page
  loadExpenses()
}

const editExpense = (expense: Expense) => {
  router.push(`/expenses/edit/${expense.id}`)
}

const confirmDelete = (expense: Expense) => {
  selectedExpense.value = expense
  showDeleteDialog.value = true
}

const handleDelete = async () => {
  if (!selectedExpense.value) return
  
  try {
    await expenseStore.deleteExpense(selectedExpense.value.id)
    showDeleteDialog.value = false
    selectedExpense.value = null
    await loadExpenses() // 一覧を再読み込み
  } catch (error) {
    console.error('Failed to delete expense:', error)
  }
}

const exportExpenses = async () => {
  try {
    const blob = await expenseStore.exportExpenses(searchCriteria.value)
    const url = URL.createObjectURL(blob)
    const a = document.createElement('a')
    a.href = url
    a.download = `expenses_${new Date().toISOString().split('T')[0]}.csv`
    a.click()
    URL.revokeObjectURL(url)
  } catch (error) {
    console.error('Failed to export expenses:', error)
  }
}
</script>
```

### 保存された検索条件
```typescript
// composables/useSavedSearches.ts
export const useSavedSearches = () => {
  const savedSearches = ref<SavedSearch[]>([])
  
  const saveSearch = (name: string, criteria: ExpenseSearchCriteria) => {
    const savedSearch: SavedSearch = {
      id: generateId(),
      name,
      criteria,
      createdAt: new Date()
    }
    
    savedSearches.value.push(savedSearch)
    localStorage.setItem('savedSearches', JSON.stringify(savedSearches.value))
  }
  
  const loadSavedSearches = () => {
    const saved = localStorage.getItem('savedSearches')
    if (saved) {
      savedSearches.value = JSON.parse(saved)
    }
  }
  
  return {
    savedSearches,
    saveSearch,
    loadSavedSearches
  }
}
```

### 検索結果のキャッシュ
```typescript
// composables/useExpenseCache.ts
const cache = new Map<string, CachedResult>()

export const useExpenseCache = () => {
  const getCachedResult = (criteria: ExpenseSearchCriteria) => {
    const key = JSON.stringify(criteria)
    const cached = cache.get(key)
    
    if (cached && Date.now() - cached.timestamp < 5 * 60 * 1000) { // 5分間有効
      return cached.data
    }
    
    return null
  }
  
  const setCachedResult = (criteria: ExpenseSearchCriteria, data: any) => {
    const key = JSON.stringify(criteria)
    cache.set(key, {
      data,
      timestamp: Date.now()
    })
  }
  
  return {
    getCachedResult,
    setCachedResult
  }
}
```
