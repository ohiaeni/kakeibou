### 2. バックアップ確認

#### データベースバックアップ
```bash
#!/bin/bash
# daily_backup_check.sh

BACKUP_DATE=$(date +%Y%m%d)
BACKUP_FILE="backup_${BACKUP_DATE}.sql.gz"

# バックアップファイルの存在確認
if [ -f "/backup/${BACKUP_FILE}" ]; then
    echo "✅ バックアップファイル確認: ${BACKUP_FILE}"
    
    # ファイルサイズ確認（最小サイズ100MB）
    SIZE=$(stat -c%s "/backup/${BACKUP_FILE}")
    MIN_SIZE=104857600  # 100MB
    
    if [ $SIZE -gt $MIN_SIZE ]; then
        echo "✅ バックアップサイズ正常: $(($SIZE / 1024 / 1024))MB"
    else
        echo "❌ バックアップサイズ異常: $(($SIZE / 1024 / 1024))MB"
        # アラート送信
        send_alert "バックアップサイズ異常" "$BACKUP_FILE"
    fi
else
    echo "❌ バックアップファイル未作成: ${BACKUP_FILE}"
    send_alert "バックアップ失敗" "$BACKUP_FILE"
fi
```

### 3. パフォーマンス監視

#### レスポンス時間チェック
```bash
#!/bin/bash
# performance_check.sh

API_ENDPOINTS=(
    "/api/v1/auth/login"
    "/api/v1/expenses"
    "/api/v1/categories"
)

for endpoint in "${API_ENDPOINTS[@]}"; do
    RESPONSE_TIME=$(curl -w "%{time_total}" -s -o /dev/null http://localhost:8080$endpoint)
    echo "Endpoint: $endpoint, Response Time: ${RESPONSE_TIME}s"
    
    # 1秒以上の場合はアラート
    if (( $(echo "$RESPONSE_TIME > 1.0" | bc -l) )); then
        send_alert "レスポンス時間超過" "$endpoint: ${RESPONSE_TIME}s"
    fi
done
```

### 3. 容量計画

```sql
-- データ増加率分析
WITH monthly_growth AS (
    SELECT 
        DATE_TRUNC('month', expense_date) as month,
        COUNT(*) as record_count
    FROM expense_records 
    WHERE expense_date >= NOW() - INTERVAL '12 months'
    GROUP BY DATE_TRUNC('month', expense_date)
    ORDER BY month
)
SELECT 
    month,
    record_count,
    LAG(record_count) OVER (ORDER BY month) as prev_month,
    ROUND(
        (record_count - LAG(record_count) OVER (ORDER BY month)) * 100.0 / 
        LAG(record_count) OVER (ORDER BY month), 
        2
    ) as growth_rate_percent
FROM monthly_growth;
```

### 3. 障害対応フロー

#### 初期対応（15分以内）
```bash
#!/bin/bash
# incident_response.sh

echo "=== 障害対応開始 ==="
echo "時刻: $(date)"

# 1. システム状況確認
echo "1. システム状況確認"
curl -s http://localhost:8080/actuator/health | jq .

# 2. プロセス確認
echo "2. プロセス確認"
ps aux | grep java

# 3. リソース使用状況
echo "3. リソース使用状況"
free -m
df -h

# 4. 直近のエラーログ
echo "4. 直近のエラーログ"
tail -100 /app/logs/application.log | grep ERROR

# 5. 緊急対応判定
if ! curl -s http://localhost:8080/actuator/health > /dev/null; then
    echo "❌ アプリケーション応答なし - 緊急対応実行"
    
    # アプリケーション再起動
    systemctl restart kakeibou-app
    
    # 再起動後の確認
    sleep 30
    if curl -s http://localhost:8080/actuator/health > /dev/null; then
        echo "✅ アプリケーション復旧"
        send_notification "アプリケーション復旧完了"
    else
        echo "❌ 復旧失敗 - エスカレーション"
        escalate_incident
    fi
fi
```

### 4. 復旧手順

#### データベース復旧
```bash
#!/bin/bash
# database_recovery.sh

# 1. データベース接続確認
if ! psql -h localhost -U kakeibou_user -d kakeibou_prod -c "SELECT 1;" > /dev/null 2>&1; then
    echo "❌ データベース接続失敗"
    
    # 2. PostgreSQLサービス確認
    systemctl status postgresql
    
    # 3. PostgreSQL再起動
    systemctl restart postgresql
    
    # 4. 復旧確認
    sleep 10
    if psql -h localhost -U kakeibou_user -d kakeibou_prod -c "SELECT 1;" > /dev/null 2>&1; then
        echo "✅ データベース復旧"
    else
        echo "❌ データベース復旧失敗 - バックアップからの復元が必要"
        restore_from_backup
    fi
fi
```

#### バックアップからの復元
```bash
#!/bin/bash
# restore_from_backup.sh

BACKUP_DATE=$(date -d "1 day ago" +%Y%m%d)
BACKUP_FILE="backup_${BACKUP_DATE}.sql.gz"

echo "バックアップからの復元開始: ${BACKUP_FILE}"

# 1. メンテナンスモード有効化
enable_maintenance_mode

# 2. バックアップファイル確認
if [ ! -f "/backup/${BACKUP_FILE}" ]; then
    echo "❌ バックアップファイルが見つかりません"
    exit 1
fi

# 3. データベース復元
gunzip -c "/backup/${BACKUP_FILE}" | psql -h localhost -U kakeibou_user -d kakeibou_prod

# 4. 整合性確認
psql -h localhost -U kakeibou_user -d kakeibou_prod -c "
    SELECT 
        COUNT(*) as user_count 
    FROM user_accounts;
    
    SELECT 
        COUNT(*) as expense_count,
        MAX(expense_date) as latest_expense
    FROM expense_records;
"

# 5. アプリケーション再起動
systemctl restart kakeibou-app

# 6. メンテナンスモード解除
disable_maintenance_mode

echo "✅ 復元完了"
```

### 2. 証明書更新

```bash
#!/bin/bash
# ssl_certificate_renewal.sh

# Let's Encrypt証明書の自動更新
certbot renew --dry-run

# 証明書の有効期限確認
openssl x509 -in /etc/ssl/certs/kakeibou.crt -text -noout | grep "Not After"

# 有効期限が30日以内の場合は警告
EXPIRY_DATE=$(openssl x509 -in /etc/ssl/certs/kakeibou.crt -text -noout | grep "Not After" | awk '{print $4, $5, $7}')
EXPIRY_TIMESTAMP=$(date -d "$EXPIRY_DATE" +%s)
CURRENT_TIMESTAMP=$(date +%s)
DAYS_UNTIL_EXPIRY=$(( ($EXPIRY_TIMESTAMP - $CURRENT_TIMESTAMP) / 86400 ))

if [ $DAYS_UNTIL_EXPIRY -le 30 ]; then
    send_alert "SSL証明書期限警告" "証明書の有効期限まで${DAYS_UNTIL_EXPIRY}日です"
fi
```

## 運用自動化

### 1. 自動化スクリプト

#### システム監視自動化
```bash
#!/bin/bash
# automated_monitoring.sh

# crontab設定例:
# */5 * * * * /opt/kakeibou/scripts/automated_monitoring.sh

LOG_FILE="/var/log/kakeibou-monitoring.log"

{
    echo "=== $(date) ==="
    
    # ヘルスチェック
    if curl -f http://localhost:8080/actuator/health > /dev/null 2>&1; then
        echo "✅ Application healthy"
    else
        echo "❌ Application unhealthy"
        systemctl restart kakeibou-app
    fi
    
    # ディスク使用量チェック（80%超過でアラート）
    DISK_USAGE=$(df / | awk 'NR==2 {print $5}' | sed 's/%//')
    if [ $DISK_USAGE -gt 80 ]; then
        echo "❌ Disk usage: ${DISK_USAGE}%"
        send_alert "ディスク使用量警告" "使用量: ${DISK_USAGE}%"
    else
        echo "✅ Disk usage: ${DISK_USAGE}%"
    fi
    
    echo ""
} >> $LOG_FILE
```
